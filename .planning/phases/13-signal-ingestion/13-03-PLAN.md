    ```python
    def _fetch_and_store_signal(
        signal_name: str,
        fetch_fn,
        cache_filename: str,
        api_key: str,
        ttl_hours: int,
        **context
    ) -> list[str]:
        """Fetch a prediction signal, cache it, and return unavailable match IDs.
        
        Implements D-07/D-08/D-09: graceful degradation, aggregated warnings.
        
        Args:
            signal_name: Display name for warnings (e.g., "market_odds", "catboost").
            fetch_fn: Callable that takes (api_key, **context) and returns cache dict.
            cache_filename: Cache filename in data/ directory.
            api_key: BSD API key.
            ttl_hours: Cache TTL in hours.
            **context: Additional keyword arguments passed to fetch_fn.
        
        Returns:
            List of warning strings (empty if all OK).
        """
        import logging
        logger = logging.getLogger(__name__)
        cache = load_signal_cache(cache_filename)
        warnings: list[str] = []
        
        if not is_cache_valid(cache, ttl_hours):
            try:
                new_cache = fetch_fn(api_key=api_key, **context)
                save_signal_cache(new_cache, cache_filename)
                cache = new_cache
            except Exception as e:
                logger.warning("%s fetch failed: %s", signal_name, e)
                if not cache:
                    warnings.append(f"{signal_name} unavailable — no cached data")
                    return warnings
        
        # Count unavailable matches
        matches = cache.get("matches", {})
        unavailable = [
            mid for mid, m in matches.items()
            if not m.get("available", False)
        ]
        if unavailable:
            warnings.append(
                f"⚠ {signal_name} unavailable for {len(unavailable)} match(es)"
            )
        
        return warnings
    ```
